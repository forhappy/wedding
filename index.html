include "父母.h"
include "亲戚.h"
include "朋友.h"
include "新娘/王欣.h"
include "新郎/傅海平.h"

// 幸福婚礼开始.
int main(int argc, const char *argv[]) {
时间 today = new 时间("2014年10月5日上午11时58分");
时间 forever = new 时间("永远");
父母 family = new 父母("宋建琴", "李复原", "王松芝", "傅松林");
新郎 i = new 新郎("傅海平");
新娘 u = new 新娘("王欣");
i.love(u);// 宝贝, 我 ♥ 你.
family.supportAndBlessUs(i, u);// 我们得到了家长和朋友的支持和祝福.
i.marry(u, family, today);// 在家人和朋友的见证下，我们结婚啦!
// 幸福大循环.
while (true) {
i.liveHappilyWith(u, today, forever);// 从此刻开始我们过上了幸福美满的生活!
}
}
